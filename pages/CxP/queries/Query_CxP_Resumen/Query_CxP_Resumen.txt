SELECT 
  cxp.id_suplidor,
  cxp.razon_social,

  MAX(cxp.codigo_iso) AS codigo_iso,

  SUM(cxp.monto_total) AS monto_total,
  SUM(cxp.balance) AS balance,

  SUM(ISNULL(cxp.[0_30],0)) AS rango_0_30,
  SUM(ISNULL(cxp.[31_60],0)) AS rango_31_60,
  SUM(ISNULL(cxp.[61_90],0)) AS rango_61_90,
  SUM(ISNULL(cxp.[90_mas],0)) AS rango_90_mas,

  ISNULL(MAX(cred.creditos_disponibles),'') AS creditos_disponibles

FROM vw_cuentas_por_pagar cxp

LEFT JOIN
(
    SELECT
        id_proyecto,
        id_suplidor,

        STRING_AGG(
            codigo_iso + ' ' +
            FORMAT(credito_disponible, 'N2'),
            ' | '
        ) AS creditos_disponibles

    FROM vw_resumen_creditos_suplidores

    GROUP BY
        id_proyecto,
        id_suplidor

) cred

    ON cred.id_proyecto = cxp.id_proyecto
   AND cred.id_suplidor = cxp.id_suplidor

WHERE 
  cxp.id_proyecto = {{ Number(Seleccionar_Proyecto_Resumen.selectedOptionValue) }}

  AND (
    {{Seleccionar_Moneda_Resumen.selectedOptionValue}} IS NULL
    OR cxp.id_moneda = {{ Number(Seleccionar_Moneda_Resumen.selectedOptionValue) }}
  )

GROUP BY 
  cxp.id_suplidor,
  cxp.razon_social

ORDER BY 
  balance DESC
